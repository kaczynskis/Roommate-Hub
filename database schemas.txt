user {
    username {
        PRIMARY KEY,
        type: VARCHAR,
        length: 50
    },
    password {
        type: VARCHAR,
        length: 50
    },
    room {
        type: VARCHAR,
        length: 50
    }
}

chores {
    id {
        PRIMARY KEY,
	    AUTO_INCREMENT
        type: int,
        length: 11
    },
    chorename {
        type: VARCHAR,
        length: 50
    },
    choredescription {
        type: VARCHAR,
        length: 140
    },
    assignedto {
        FOREIGN KEY REFERENCES user(username),
        type: VARCHAR,
        length: 50
    }
}

payments {
    id {
        PRIMARY KEY,
        AUTO_INCREMENT
        type: int,
        length: 11
    },
    amount {
        type: VARCHAR,
        length: 10
    },
    borrower {
        FOREIGN KEY REFERENCES user(username),
        type: VARCHAR,
        length: 50
    },
    owedto {
        FOREIGN KEY REFERENCES user(username),
        type: VARCHAR,
        length: 50
    },
    paymentDescription {
	type: VARCHAR,
	length: 50
    }
}
groceries {
    id {
        PRIMARY KEY,
	    AUTO_INCREMENT
        type: int,
        length: 11
    },
    itemname {
        type: VARCHAR,
        length: 50
    },
    itemdescription {
        type: VARCHAR,
        length: 140
    },
    itemcount {
        type: INT,
        length: 5
    }
    requestedby {
        FOREIGN KEY REFERENCES user(username),
        type: VARCHAR,
        length: 50
    }
}
